  minutes: number
  words: number
}

export type BlogPost = {
  id: string
  slug: string
  title: string
  subtitle: string
  preview: string           // 2-3 sentence teaser
  content: string           // full post in markdown
  tags: string[]
  category: "Debug" | "Tutorial" | "Project" | 
            "Research" | "Opinion" | "Journey"
  status: BlogStatus
  publishedAt: string       // "DD MMM YYYY" or ""
  updatedAt?: string
  readTime: ReadTime
  featured: boolean
  series?: {                // if part of a series
    name: string
    part: number
    total: number
  }
}

    Active tag chip: filled --accent bg
